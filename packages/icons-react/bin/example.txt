# 快速上手

## 引用组件

```
yarn add @yaner/icons-react
```

## 引入组件样式

```
import React from 'react';
import { SmilingOutlined } from '@yaner/icons-react';
```

## 使用组件 [React/Vue]

```
import React from 'react';
import { SmilingOutlined } from '@yaner/icons-react';

const Wrapper: React.SF = () => {
  return (
  	<div>
    	<SmilingOutlined style={{ fontSize: '32px', color: '#fff' }}/>
    </div>
  )
}

export default Wrapper
```
