# 快速上手

## 引用组件

```
yarn add @yee/icons-react
```

## 引入组件样式

```
import React from 'react';
import { SmilingOutlined } from '@yee/icons-react';
```

## 使用组件

```
import React from 'react';
import { SmilingOutlined } from '@yee/icons-react';

const Wrapper: React.SF = () => {
  return (
  	<div>
    	<SmilingOutlined />
    </div>
  )
}

export default Wrapper
```
